13 void MappingThreader::thread_mappings() {
14  sglib::OutputLog() << "Assembling mappings into contiguous threads" << std::endl;
15 
16  FastaReader<FastaRecord> fastareader({0}, getQueryFile());
17  FastaRecord refsequence;
18  std::vector<KmerIDX> ref_seq_kmers;
19  kmerIDXFactory<FastaRecord> kmerfactory({ getK() });
20  PathExplorer pe { getGraph() };
21  MappingThread mappingThread(getGraph());
22  std::vector<MappingThread> mappingThreads;
23 
24  const std::unordered_map<seqID_t, std::vector<NodeMapping>> &allMappings { mapper.getMappings() };
25 
26  bool c;
27  c = fastareader.next_record(refsequence);
28 
29  while (c) {
30  sglib::OutputLog() << "Threading mappings for " << refsequence.name << std::endl;
31  const auto &mappings = allMappings.at(refsequence.id);
32 
33  for(const auto &sm : mappings) {
34  // For every mapping hit the sequence has, try to append the node of the mapping hit to the current path.
35  sglib::OutputLog(sglib::DEBUG) << "Considering the mapping:" << std::endl << getGraph().nodeID_to_name(sm.absnode()) << std::endl;
36  sglib::OutputLog(sglib::DEBUG) << "Trying to append mapping to current thread trivially" << std::endl;
37  // There are a few cases that could happen here. First, either the mapping could be appended, trivially.
38  bool could_append_trivially = append_mapping_trivially(mappingThread, sm);
39  if(could_append_trivially) {
40  sglib::OutputLog(sglib::DEBUG) << "Appended mapping to current thread" << std::endl;
41  } else {
42  sglib::OutputLog(sglib::DEBUG) << "Failed to append mapping to current thread trivially" << std::endl;
43  sglib::OutputLog(sglib::DEBUG) << "Trying to append mapping to current thread by path finding" << std::endl;
44 
45  // Work out the region of the sub sequence to extract from the reference sequence...
46  auto first_idx = mappingThread.lastMapping().query_start();
47  auto final_idx = sm.query_end();
48  if (first_idx > final_idx) {
49  std::swap(first_idx, final_idx);
50  }
51  // Extract the sub-sequence from the reference sequence...
52  size_t len { size_t(final_idx) - size_t(first_idx) + 1 };
53  std::string refsubstr { refsequence.seq.substr((size_t) first_idx, len) };
54 
55  SequenceGraphPath bestpath { getGraph() };
56  bool found_path = pe.find_best_path(bestpath, mappingThread.lastMapping().dirnode(), sm.dirnode(), refsubstr);
57 
58  if (found_path) {
59  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Found a suitable path connecting the two mappings" << std::endl;
60  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Adding the nodes of this suitable path to the sequence mapping thread" << std::endl;
61  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Current bestpath: ";
62  if (sglib::OutputLogLevel >= sglib::LogLevels::DEBUG) {
63  for (const auto &node : bestpath.getNodes()) {
64  std::cout << node << ", ";
65  }
66  std::cout << std::endl;
67  }
68  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Adding nodes of bestpath to the current thread:" << std::endl;
69  append_mapping_with_path(mappingThread, sm, bestpath);
70  } else {
71  // What happens if you couldn't append trivially, AND could not append by searching a path.
72  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Could not trivially link mappings, or get there through path search" << std::endl;
73  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Saving current path." << std::endl;
74  mappingThreads.emplace_back(mappingThread);
75  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Clearing path to start a new path." << std::endl;
76  mappingThread.clear();
77  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Adding node to new path." << std::endl;
78  append_mapping_trivially(mappingThread, sm);
79  }
80  }
81  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Current path of mappings is " << mappingThread.size() << " nodes long." << std::endl;
82  }
83  sglib::OutputLog() << "Finished connecting paths for sequence " << refsequence.name << std::endl;
84  mappingThreads.emplace_back(mappingThread);
85  mappingThread.clear();
86  mapping_threads_of_sequence.emplace(refsequence.id, mappingThreads);
87  mappingThreads.clear();
88  c = fastareader.next_record(refsequence);
89  }
90 }

139 bool MappingThreader::append_mapping_trivially(MappingThread &thread, NodeMapping mapping) {
140  auto dn = mapping.dirnode();
141  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Trying to add to path trivially as: " << dn << std::endl;
142  auto pathSuccess = thread.nodePath().append_to_path(dn);
143  if (pathSuccess) {
144  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Was able to trivially append " << dn << " to the path." << std::endl;
145  thread.orderedMappings().emplace_back(mapping);
146  } else {
147  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Was not able to trivially append " << dn << " to the path." << std::endl;
148  }
149  return pathSuccess;
150 }
