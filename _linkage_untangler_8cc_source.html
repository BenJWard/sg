196 std::set<std::pair<sgNodeID_t, sgNodeID_t >> LinkageUntangler::get_HSPNPs(uint64_t min_size, float min_ci,
197  float max_ci) {
198  std::set<std::pair<sgNodeID_t, sgNodeID_t >> hspnps;
199  SequenceGraph& sg(ws.getGraph());
200 #pragma omp parallel for schedule(static, 100)
201  for (sgNodeID_t n = 1; n < sg.nodes.size(); ++n) {
202  if (sg.nodes[n].status == sgNodeDeleted) continue;
203  if (sg.nodes[n].sequence.size() < min_size) continue;
204  //FW check
205  auto fwl = sg.get_fw_links(n);
206  if (fwl.size() != 1) continue;
207  auto post = fwl[0].dest;
208  auto post_bwl = sg.get_bw_links(post);
209  if (post_bwl.size() != 2) continue;
210  if (llabs(post_bwl[0].dest)==llabs(post_bwl[1].dest))continue;
211  //BW check
212  auto bwl = sg.get_bw_links(n);
213  if (bwl.size() != 1) continue;
214  auto prev = bwl[0].dest;
215  auto prev_fwl = sg.get_bw_links(prev);
216  if (prev_fwl.size() != 2) continue;
217 
218  if ((prev_fwl[0].dest == -post_bwl[0].dest and prev_fwl[1].dest == -post_bwl[1].dest)
219  or (prev_fwl[1].dest == -post_bwl[0].dest and prev_fwl[0].dest == -post_bwl[1].dest)) {
220  sgNodeID_t m;
221  if (llabs(prev_fwl[0].dest) != n and llabs(prev_fwl[1].dest) != n) std::cout<<"Error! cant find N in prev!"<<std::endl;
222  if (llabs(prev_fwl[0].dest) == n) m = llabs(prev_fwl[1].dest);
223  else m = prev_fwl[0].dest;
224  //Now evaluate coverage of the branches
225  auto c1 = ws.getKCI().compute_compression_for_node(n, 1);
226  if (std::isnan(c1) or c1<min_ci or c1>max_ci) continue;
227  auto c2 = ws.getKCI().compute_compression_for_node(m, 1);
228  if (std::isnan(c2) or c2<min_ci or c2>max_ci) continue;
229 #pragma omp critical(inserting_hspnps)
230  {
231  //hl<<(n<m ? n:m)<<" "<<(n<m ? m:n)<<std::endl;
232  if (n < llabs(m)) hspnps.insert(std::make_pair(n, m));
233  else hspnps.insert(std::make_pair(llabs(m), (m>0 ? n:-n)));
234  }
235  }
236  }
237  return hspnps;
238 }
