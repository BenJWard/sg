114  inline void produce_all_kmers(const char * seq, std::vector<uint64_t> &mers){
115  // TODO: Adjust for when K is larger than what fits in uint64_t!
116  last_unknown=0;
117  fkmer=0;
118  rkmer=0;
119  auto s=seq;
120  while (*s!='\0' and *s!='\n') {
121  //fkmer: grows from the right (LSB)
122  //rkmer: grows from the left (MSB)
123  fillKBuf(*s, fkmer, rkmer, last_unknown);
124  if (last_unknown >= K) {
125  if (fkmer <= rkmer) {
126  // Is fwd
127  mers.emplace_back(fkmer);
128  } else {
129  // Is bwd
130  mers.emplace_back(rkmer);
131  }
132  }
133  ++s;
134  }
135  }
136  };
