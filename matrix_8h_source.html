92  // resizing
93  void resize(size_type new_rows, size_type new_cols) {
94  // new matrix new_rows times new_cols
95  matrix tmp(new_rows, new_cols);
96  // select smaller row and col size
97  auto mc = std::min(m_cols, new_cols);
98  auto mr = std::min(m_rows, new_rows);
99  for (size_type i(0U); i < mr; ++i) {
100  // iterators to begin of rows
101  auto row = begin() + i * m_cols;
102  auto tmp_row = tmp.begin() + i * new_cols;
103  // move mc elements to tmp
104  std::move(row, row + mc, tmp_row);
105  }
106  // move assignment to this
107  *this = std::move(tmp);
108  }
