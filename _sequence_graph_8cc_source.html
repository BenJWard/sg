25 bool Node::is_canonical() {
26  for (size_t i=0,j=sequence.size()-1;i<j;++i,--j){
27  char f=sequence[i];
28  char r=sequence[j];
29  switch(r){
30  case 'A':
31  r='T';
32  break;
33  case 'C':
34  r='G';
35  break;
36  case 'G':
37  r='C';
38  break;
39  case 'T':
40  r='A';
41  break;
42  case 'N':
43  break;
44  default:
45  sglib::OutputLog(sglib::LogLevels::WARN) << "Unexpected character in fasta file: '" << r << "'" << std::endl;
46  }
47  if (f<r) return true;
48  if (r<f) return false;
49  }
50  return true;

511  }

515  }

663  to_visit.pop();

668  (activeNode.dist < size_limit or size_limit==0) ) // Is within sequence limits

681  }

734 SequenceGraphPath
735 SequenceGraph::find_path_between(const sgNodeID_t seed, const sgNodeID_t target, unsigned int size_limit,
736  unsigned int edge_limit) {
737  struct visitor {
738  sgNodeID_t node;
739  uint dist;
740  uint path_length;
741  visitor(sgNodeID_t n, uint d, uint p) : node(n), dist(d), path_length(p) {}
742  };
743 
744  std::vector<SequenceGraphPath> collected_paths;
745  std::vector<sgNodeID_t> current_path;
746 
747  std::stack<visitor> to_visit;
748  to_visit.emplace(seed, 0, 0);
749  std::set<sgNodeID_t> visited;
750 
751  while (!to_visit.empty()) {
752  const auto activeNode(to_visit.top());
753  to_visit.pop();
754  if (visited.find(activeNode.node) == visited.end() and (activeNode.path_length < edge_limit or edge_limit==0) and (activeNode.dist < size_limit or size_limit==0) )
755  {
756  visited.emplace(activeNode.node);
757 
758  if (current_path.size() > activeNode.path_length) {
759  current_path[activeNode.path_length] = activeNode.node;
760  current_path.resize(activeNode.path_length + 1);
761  } else {
762  current_path.emplace_back(activeNode.node);
763  }
764 
765  for (const auto &l : get_fw_links(activeNode.node)) {
766  // For each child of the current node, create a child visitor object, and enter it into the parent map.
767  visitor child { l.dest, activeNode.dist + uint(nodes[std::abs(l.dest)].sequence.length()), activeNode.path_length + 1 };
768 
769  // If I've found the target along this dfs path, I need to collect it and put it into a SequenceGraphPath object.
770  if (child.node == target) {
771  current_path.emplace_back(target);
772  return SequenceGraphPath(*this,current_path);
773  } else {
774  to_visit.emplace(child);
775  }
776  }
777  }
778  }
779  return SequenceGraphPath{*this};
780 }

783 SequenceGraph::explore_nodes(std::vector<std::string> &nodes, unsigned int size_limit, unsigned int edge_limit) {

786  for (const auto &n:nodes) {
787  auto id = oldnames_to_ids.at(n);
788  std::set<nodeVisitor> results;
789  results.emplace(id, 0, 0);
790  results.emplace(-id, 0, 0);

797  sglib::OutputLog(sglib::DEBUG,false) << std::endl;
798  auto toVisit = *results.begin();
799  results.erase(toVisit);
800  // Explore node forward
801  auto visitedNodes = depth_first_search(toVisit, size_limit, edge_limit, resultNodes);
802 
803  // For each visited node
804  for (const auto &node:visitedNodes) {
805  // If is the same as the node I am visiting, do nothing
806  if (node == toVisit) continue;
807 
808  auto visitedNode(resultNodes.find(node));
809 
810  // If was in previous results
811  if (visitedNode != resultNodes.end()) {
812  // If now has a shorter path or closer distance

814  resultNodes.erase(visitedNode);
815  } else {
816  continue;
817  }
818  }
819  results.insert(node);
820  nodeVisitor rev = nodeVisitor(-1*node.node, node.dist, node.path_length);
821  results.insert(rev);
822  resultNodes.emplace(node);
823  }
824  // While exploration results > 0 explore resulting nodes
825  } while (!results.empty());
826  }
827  return std::vector<nodeVisitor>(resultNodes.begin(), resultNodes.end());

912  /*
913  * the loop always keep the first and the last elements as c=2 collapsed nodes.
914  * it starts with a c=2 node, and goes thorugh all bubbles fw, then reverts the subgraph and repeats

916  SequenceSubGraph subgraph(*this);

924  for (auto pass=0; pass<2; ++pass) {
925  //while there's a possible bubble fw.
926  for (auto fn = get_fw_links(subgraph.nodes.back()); fn.size() == 2; fn = get_fw_links(subgraph.nodes.back())) {
927  //if it is not a real bubble, get out.
928  if (get_bw_links(fn[0].dest).size()!=1 or get_bw_links(fn[0].dest).size()!=1) break;
929  auto fl1=get_fw_links(fn[0].dest);
930  if (fl1.size()!=1) break;
931  auto fl2=get_fw_links(fn[1].dest);

935  //all conditions met, update subgraph
936  if (used[llabs(fn[0].dest)] or used[llabs(fn[1].dest)] or used[llabs(next_end)]) {
937  circular=true;
938  break;
939  }

941  subgraph.nodes.push_back(fn[1].dest);
942  subgraph.nodes.push_back(next_end);
