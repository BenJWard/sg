8 void LongReadMapper::map_reads(std::unordered_set<uint32_t> readIDs) {
9  std::vector<std::vector<LongReadMapping>> thread_mappings(omp_get_max_threads());
10 #pragma omp parallel
11  {
12  mm_tbuf_t *buf = mm_tbuf_init();
13 #pragma omp for
14  for (uint32_t readID = 1; readID < datastore.size(); ++readID) {
15  if (!((readIDs.size()>0 and readIDs.count(readID)>0) or (readIDs.empty() and read_to_mappings[readID].empty())))
16  continue;
17 
18  std::string read_seq(datastore.get_read_sequence(readID));
19  std::string read_name(std::to_string(readID));
20  int read_len(static_cast<int>(read_seq.length()));
21  int n_regs0;
22  mm_reg1_t *regs0 = mm_map(graph_index, read_len, read_seq.data(), &n_regs0, buf, &opt, read_name.data());
23 
24  if (n_regs0<=1) {
25  for (int j = 0; j < n_regs0; j++) {
26  auto node = regs0[j].rid + 1;
27  LongReadMapping mapping = createMapping(readID, regs0, j, node);
28  thread_mappings[omp_get_thread_num()].emplace_back(mapping);
29  }
30  }
31  if (n_regs0 > 1) {
32  for (int j = 0; j < n_regs0 - 1; ++j) {
33  auto fromNode = regs0[j].rid+1;
34  auto toNode = regs0[j+1].rid+1;
35  LongReadMapping fromMapping = createMapping(readID, regs0, j, fromNode);
36  LongReadMapping toMapping = createMapping(readID, regs0, j+1, toNode);
37 
38  if (link_is_valid(fromMapping, toMapping)) {
39  thread_mappings[omp_get_thread_num()].emplace_back(fromMapping);
40  thread_mappings[omp_get_thread_num()].emplace_back(toMapping);
41  }
42  }
43  }
44  for (int i = 0; i<n_regs0;i++) free(regs0[i].p);
45  free(regs0);
46  }
47  mm_tbuf_destroy(buf);
48  }
49  for (int thread = 0; thread<omp_get_max_threads(); thread++) {
50  mappings.reserve(thread_mappings.size());
51  for (const auto &mapping : thread_mappings[thread]) {
52  mappings.emplace_back(mapping);
53  }
54  }
55 
56  update_indexes_from_mappings();
57 }

59 LongReadMapping LongReadMapper::createMapping(uint32_t readID, const mm_reg1_t *regs0, int j, long long int node) const {
60  LongReadMapping mapping;
61  mapping.node = node * ((regs0[j].rev == 0) ? 1 : -1);
62  mapping.nStart = regs0[j].rs;
63  mapping.nEnd = regs0[j].re;
64  mapping.qStart = regs0[j].qs;
65  mapping.qEnd = regs0[j].qe;
66  mapping.read_id = readID;
67  mapping.score = regs0[j].score;
68  return mapping;
69 }

129 bool LongReadMapper::link_is_valid(const LongReadMapping &fromMapping, const LongReadMapping &toMapping) {
130  // Check that the mappings are larger than 200bp
131  if (fromMapping.qEnd-fromMapping.qStart < 80 or toMapping.qEnd-toMapping.qStart < 80) {
132  return false;
133  }
134  // Check the query is not overlapped by more than 60%
135  if (toMapping.qStart <= fromMapping.qEnd) {
136  auto olpSize = std::max(fromMapping.qStart,toMapping.qStart) - std::min(fromMapping.qEnd,toMapping.qEnd);
137  if (olpSize > std::abs(fromMapping.qEnd - fromMapping.qStart)*.8 or olpSize > std::abs(toMapping.qEnd-toMapping.qStart)*.8) {
138  return false;
139  }
140  }
141 
142  // Check that the score is over 90% on both ends
143 // if (fromMapping.score < std::abs(fromMapping.qEnd-fromMapping.qStart)*.9f or
144 // toMapping.score < std::abs(toMapping.qEnd - toMapping.qStart)*.9f) {
145 // return false;
146 // }
147 
148  return true;
149 }
