186 std::vector<std::pair<sgNodeID_t,sgNodeID_t>> Scaffolder::get_all_haplotype_pairs(uint32_t maxpairs) {
187  std::vector<std::pair<sgNodeID_t,sgNodeID_t>> hps;
188  std::vector<bool> used(sg.nodes.size(),false);
189  //TODO: check the coverages are actually correct?
190  const double min_c1=0.7,max_c1=1.30,min_c2=1.8,max_c2=2.8;
191  /*
192  * the loop always keep the first and the last elements as c=2 collapsed nodes.
193  * it starts with a c=2 node, and goes thorugh all bubbles fw, then reverts the subgraph and repeats
194  */
195  std::pair<sgNodeID_t,sgNodeID_t> hap={0,0};
196  for (auto n=1;n<sg.nodes.size();++n){
197  if (sg.nodes[n].status==sgNodeDeleted) continue;
198  auto frontkci=kci.compute_compression_for_node(n);
199  if (frontkci>max_c2 or frontkci<min_c2) continue;
200  auto m=n;
201  //two passes: 0->fw, 1->bw,
202  for (auto pass=0; pass<2; ++pass,m=-m) {
203 
204  //check bubble going forward ------------
205  auto fw_l = sg.get_fw_links(m);
206  //fork opening
207  if (fw_l.size() != 2) continue;
208  hap.first = fw_l[0].dest;
209  hap.second = fw_l[1].dest;
210  if (hap.first == n or hap.first == -n or hap.second == n or hap.second == -n or hap.first == hap.second) continue;
211  //fork.closing
212  auto hap0f = sg.get_fw_links(hap.first);
213  auto hap1f = sg.get_fw_links(hap.second);
214  if (hap0f.size() != 1 or hap1f.size() != 1 or hap0f[0].dest != hap1f[0].dest) continue;
215  auto h0kc = kci.compute_compression_for_node(hap.first);
216  if (h0kc > max_c1 or h0kc < min_c1) continue;
217  auto h1kc = kci.compute_compression_for_node(hap.second);
218  if (h1kc > max_c1 or h1kc < min_c1) continue;
219  auto ekc = kci.compute_compression_for_node(hap0f[0].dest);
220  if (ekc > max_c2 or ekc < min_c2) continue;
221  if (used[(hap.first>0?hap.first:-hap.first)] or used[(hap.second>0?hap.second:-hap.second)]){
222  if (not used[(hap.second>0?hap.second:-hap.second)] or not used[(hap.second>0?hap.second:-hap.second)])
223  std::cout<<"WARNING: unusual use pattern on HSPNP"<<std::endl;
224  continue;
225  }
226  hps.push_back(hap);
227  if (hps.size()%100==0) std::cout<<hps.size()<<" haplotype pairs found"<<std::endl;
228  used[(hap.first>0?hap.first:-hap.first)]=true;
229  used[(hap.second>0?hap.second:-hap.second)]=true;
230  }
231  }
232  return hps;
