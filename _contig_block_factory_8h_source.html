47  template<typename T, typename T2=T>
48  struct accumulator {
49  T2 sum; // we could plug in a more accurate type for the sum
50  T S;
51  T M;
52  size_t N;
53 
54  // default constructor initializes all values
55  accumulator() : sum(0), S(0), M(0), N(0) {}
56 
57  // add another number
58  T2 operator()(const T &x) {
59  ++N;
60  sum += x;
61  T Mprev = M;
62  M += (x - Mprev) / N;
63  S += (x - Mprev) * (x - M);
64  return sum;
65  }
66 
67  T mean() const { return sum / N; }
68 
69  T variance() const { return S / (N - 1); }
70 
71  // operator<< to print the statistics to screen:
72  // denoted friend just to be able to write this inside
73  // the class definition and thus not to need to write
74  // the template specification of accumulator...
75  friend std::ostream &operator<<(std::ostream &out, const accumulator &a) {
76  if (a.N > 0)
77  out << "N\t\t\t= " << a.N << std::endl << "sum\t\t\t= " << a.sum << std::endl << "mean\t\t= "
78  << std::fixed << std::setprecision(2) << a.mean() << std::endl;
79  if (a.N > 1)
80  out << "sd\t\t\t= " << std::fixed << std::setprecision(2) << std::sqrt(a.variance()) << std::endl;
81  else
82  out << "sd\t\t\t= " << std::fixed << std::setprecision(2) << 0 << std::endl;
83  return out;

240  std::vector<Block> getBlocks(const std::vector<Match> &matches) const {
241  std::vector<Block> blocks;
242  uint32_t blkDif(0), offDif(0);
243  if (!matches.empty()) {
244  auto prev = matches.cbegin();
245  Block blk(0, prev->refPos, 0, prev->readPos, 0, 1, prev->dirContig);
246  auto curr = prev;
247  ++curr;
248  for (; curr != matches.cend(); ++curr) {
249  int32_t currOffset(curr->refPos-curr->readPos);
250  int32_t prevOffset(prev->refPos - prev->readPos);
251  if (blk.contigID == curr->dirContig
252  and std::abs(currOffset - prevOffset) < offset_limit ) {
253  blk.count++;
254  blk.setREnd(curr->refPos), blk.setQEnd(curr->readPos);
255  } else {
256  if (blk.contigID != curr->dirContig) blkDif++;
257  if (std::abs((int32_t)(curr->refPos - prev->refPos)) >= offset_limit) offDif++;
258  if (blk.count>min_kmers_to_call_match)
259  blocks.push_back(blk);
260  blk = Block(curr - matches.begin(), curr->refPos, 0, curr->readPos, 0, 1, curr->dirContig);
261  }
262  prev = curr;
263  }
264  if(blk.count > min_kmers_to_call_match)
265  blocks.push_back(blk);
266  }
267  read_block_stats << blkDif << "," << offDif; // First 2 values, missing validBlocks
268  return blocks;
269  }

271  std::vector<Match> getMatches() {
272  std::vector<Match> matches;
273  int64_t offset(0);
274  uint64_t p(0);
275  while (p < currentFileRecord.seq.size()) {
276  //fkmer: grows from the right (LSB)
277  //rkmer: grows from the left (MSB)
278  fillKBuf(currentFileRecord.seq[p], fkmer, rkmer, last_unknown);
279  p++;
280  if (last_unknown >= K) {
281  // Generate all the kmer-contig matches with Direction, Contig, Offset and readPosition
282  std::pair<KmerIDX,int> read_kmer (
283  (fkmer < rkmer) ? std::make_pair(KmerIDX(fkmer), 1) : std::make_pair(KmerIDX(rkmer), -1));
284  auto kmer = kmers.find(read_kmer.first);
285  if (kmer != kmers.end()) { // If the kmer from the read is in the unitigs
286  if (kmer->contigID * read_kmer.second > 0) {
287  offset = kmer->pos - p;
288  } else {
289  offset = kmer->pos + p;
290  }
291  matches.emplace_back(kmer->contigID * read_kmer.second, p, kmer->pos);
292  }
293  ctg_read << ((kmer != kmers.end()) ? abs(kmer->contigID) : 0) << " ";
294  }
295  }
296  // Sort matches by contig and offset
297  std::sort(matches.begin(), matches.end(), typename Match::byContigRef());
298  return matches;
299  }

362  const bool next_elemen2t(std::vector<Block> &blocks) {
363  blocks.clear();
364  std::cout << currentFileRecord.name << "\t" << currentFileRecord.seq.size() << std::endl;
365  bases+=currentFileRecord.seq.size();
366  read_length << currentFileRecord.name << "," << currentFileRecord.seq.size() << std::endl;
367  ctg_read << currentFileRecord.name << "(" << currentFileRecord.seq.size() << ")\n";
368  std::vector<Match> matches = getMatches(); // Return all matches sorted by position in the read
369  ctg_read<<std::endl;
370  std::copy(matches.begin(),matches.end(), std::ostream_iterator<Match>(sglib::OutputLog(sglib::DEBUG, false)," -> "));
371  std::cout << std::endl;
372 
373  uint64_t n(matches.size());
374 
375  std::vector<int32_t> f(n);
376  std::vector<int32_t> t(n);
377  std::vector<int32_t> p(n);
378  std::vector<int32_t> v(n);
379  int64_t st(0);
380  uint32_t max_dist_y(5000), max_dist_x(5000);
381  uint32_t bw(500);
382  uint32_t max_skip(25);
383  int64_t min_sc=40;
384  int min_cnt=4;
385  int32_t i(0), j(0);
386 
387  for (i = 0; i < n; ++i){
388  uint64_t ri = matches[i].refPos;
389  int32_t max_j = -1;
390  int32_t qi = matches[i].readPos;
391  int32_t n_skip = 0;
392  int32_t max_f = this->K;
393  int64_t min_d;
394  while (st < i && ri - matches[st].refPos > max_dist_x) ++st;
395  for (j = i - 1; j >= st; --j) {
396  int64_t dr = ri - matches[j].refPos;
397  int32_t dq = qi - (int32_t)matches[j].readPos, log_dd;
398  int32_t sc;
399  int64_t dd;
400  if ((dr == 0) || dq <= 0) continue; // don't skip if an anchor is used by multiple segments; see below
401  if ((dq > max_dist_y) || dq > max_dist_x) continue;
402  dd = dr > dq? dr - dq : dq - dr;
403  if (dd > bw) continue;
404  if (dr > max_dist_y) continue;
405  min_d = dq < dr? dq : dr;
406  sc = (int32_t) (min_d > K ? K : dq < dr ? dq : dr);
407  log_dd = dd? ilog2_32(dd) : 0;
408  sc -= (int)(dd * .01 * K) + (log_dd>>1);
409  sc += f[j];
410  if (sc > max_f) {
411  max_f = sc, max_j = j;
412  if (n_skip > 0) --n_skip;
413  } else if (t[j] == i) {
414  if (++n_skip > max_skip)
415  break;
416  }
417  if (p[j] >= 0) t[p[j]] = i;
418  }
419  f[i] = max_f, p[i] = max_j;
420  v[i] = max_j >= 0 && v[max_j] > max_f? v[max_j] : max_f; // v[] keeps the peak score up to i; f[] is the score ending at i, not always the peak
421  }
422 
423  std::fill(t.begin(), t.end(), 0);
424  // find the ending positions of chains
425  int n_u;
426  for (i = 0; i < n; ++i)
427  if (p[i] >= 0) t[p[i]] = 1;
428  for (i = n_u = 0; i < n; ++i)
429  if (t[i] == 0 && v[i] >= min_sc)
430  ++n_u;
431  if (n_u == 0) {
432  return false;
433  }
434 
435  std::vector<uint64_t > u(static_cast<unsigned long>(n_u * 8));
436  for (i = n_u = 0; i < matches.size(); ++i) {
437  if (t[i] == 0 && v[i] >= min_sc) {
438  j = i;
439  while (j >= 0 && f[j] < v[j]) j = p[j]; // find the peak that maximizes f[]
440  if (j < 0) j = i; // TODO: this should really be assert(j>=0)
441  u[n_u++] = (uint64_t)f[j] << 32 | j;
442  }
443  }
444  std::sort(u.begin(),u.begin()+n_u);
445  for (i = 0; i < n_u>>1; ++i) { // reverse, s.t. the highest scoring chain is the first
446  uint64_t t = u[i];
447  u[i] = u[n_u - i - 1], u[n_u - i - 1] = t;
448  }
449 
450  // backtrack
451  std::fill(t.begin(), t.end(), 0);
452 
453  int32_t n_v(0), k(0);
454  for (i = n_v = k = 0; i < n_u; ++i) { // starting from the highest score
455  int32_t n_v0 = n_v, k0 = k;
456  j = (int32_t)u[i];
457  do {
458  v[n_v++] = j;
459  t[j] = 1;
460  j = p[j];
461  } while (j >= 0 && t[j] == 0);
462  if (j < 0) {
463  if (n_v - n_v0 >= min_cnt) u[k++] = u[i]>>32<<32 | (n_v - n_v0);
464  } else if ((int32_t)(u[i]>>32) - f[j] >= min_sc) {
465  if (n_v - n_v0 >= min_cnt) u[k++] = ((u[i]>>32) - f[j]) << 32 | (n_v - n_v0);
466  }
467  if (k0 == k) n_v = n_v0; // no new chain added, reset
468  }
469 
470 // *n_u_ = n_u = k, *_u = u; // NB: note that u[] may not be sorted by score here
471 
472  // write the result to b[]
473  std::vector<Match> b(n_v);
474  for (i = 0, k = 0; i < n_u; ++i) {
475  int32_t k0 = k, ni = (int32_t)u[i];
476  for (j = 0; j < ni; ++j)
477  b[k] = matches[v[k0 + (ni - j - 1)]], ++k;
478  }
479 
480  // sort u[] and a[] by a[].x, such that adjacent chains may be joined (required by mm_join_long)
481  std::vector<std::pair<uint64_t, uint64_t>> w(n_u);
482  for (i = k = 0; i < n_u; ++i) {
483  w[i].first = b[k].refPos, w[i].second = (uint64_t)k<<32|i;
484  k += (int32_t)u[i];
485  }
486  std::sort(w.begin(),w.begin()+n_u);
487  std::vector<uint64_t > u2(n_u * 8);
488  std::vector<Match> validMatches;
489  for (i = k = 0; i < n_u; ++i) {
490  int32_t j = (int32_t)w[i].second, n = (int32_t)u[j];
491  u2[i] = u[j];
492  validMatches.push_back(b[w[i].second>>32]);
493  k += n;
494  }
495 
496  // Get contigs by "appearances" and "kmer matches" within 1kbp window for every 1kbp window of the read
497 
498  auto tmpTop = getTop(5, matches, currentFileRecord.seq.length(), 1000);
499 
500 // std::sort(matches.begin(),matches.rend(), typename Match::byReadPos());
501  // Generate blocks which share the same ctgOffset
502  auto tmpBlocks = getBlocks(matches);
503  std::copy(tmpBlocks.cbegin(), tmpBlocks.cend(), std::ostream_iterator<Block>(std::cout, ", "));
504  std::cout << std::endl;
505 
506 // Sort blocks by ReadPos and keep only the valid ones
507  std::copy_if(tmpBlocks.begin(), tmpBlocks.end(), std::back_inserter(blocks),
508  [&](const Block &b) { return isValid(b); });
509  std::copy(blocks.cbegin(), blocks.cend(), std::ostream_iterator<Block>(std::cout, ", "));
510  std::cout << std::endl;
511 
512 
513  // print the rest of the parameters for the blocks on this read
514  read_block_stats << "," << blocks.size() << "," << blocks.size() << std::endl;
515  std::cout << std::endl;
516  read_link_stats << blocks.size() << std::endl;
517  return false;
