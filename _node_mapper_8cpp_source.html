33  std::ifstream inf(filename, std::ios_base::binary);

36  mappings_of_sequence.reserve(mapSize);
37  seqID_t seqid;
38  std::vector<NodeMapping> vec;
39  auto vecSize { vec.size() };
40  for (decltype(mapSize) n = 0; n < mapSize; ++n){
41  inf.read(reinterpret_cast<char *>(&seqid), sizeof(seqid));
42  inf.read(reinterpret_cast<char *>(&vecSize), sizeof(vecSize));
43  vec.resize(vecSize);
44  inf.read(reinterpret_cast<char *>(vec.data()), vecSize * sizeof(NodeMapping));
45  mappings_of_sequence.emplace(seqid, vec);
46  }
47 }

49 void NodeMapper::filterMappings(double score) {
50  sglib::OutputLog(sglib::LogLevels::INFO) << "Filtering mappings using a threshold of " << score << "%" << std::endl;
51  for (auto& sequence_mappings : mappings_of_sequence) {
52  auto start { sequence_mappings.second.begin() };

104 void NodeMapper::map_sequences_from_file(const std::string &filename) {
105  sglib::OutputLog(sglib::LogLevels::INFO) << "Mapping sequence kmers to graph" << std::endl;
106  FastaReader<FastaRecord> fastaReader({0}, filename);
107  std::atomic<uint64_t> sequence_count(0);
108 
109 #pragma omp parallel shared(fastaReader, mappings_of_sequence, sequence_unmapped_kmers, query_seq_sizes, query_seq_names)
110  {
111  FastaRecord sequence;
112  bool c;
113 
114 #pragma omp critical (getrecord)
115  {
116  c = fastaReader.next_record(sequence);
117  }
118 
119  while (c) {
120  sequence_count++;
121  std::vector<NodeMapping> mappings;
122  std::vector<KmerIDX> unmapped_kmers;
123  sglib::OutputLog(sglib::LogLevels::INFO) << "Mapping sequence: " << sequence.name << " (" << sequence.id << ')' << std::endl;
124  std::tie(mappings, unmapped_kmers) = map_sequence_to_graph(sequence);
125 
126 #pragma omp critical (store)
127  {
128  sglib::OutputLog(sglib::LogLevels::INFO) << "Failed to map " << unmapped_kmers.size() << " kmers from the reference" << std::endl;
129  mappings_of_sequence[sequence.id] = mappings;
130  sequence_unmapped_kmers[sequence.id] = unmapped_kmers;
131  query_seq_sizes[sequence.id] = sequence.seq.size();
132  query_seq_names[sequence.id] = sequence.name;
133  }
134 
135 #pragma omp critical (getrecord)
136  {
137  c = fastaReader.next_record(sequence);
138  }
139  }
140  }
141 }

143 std::tuple<std::vector<NodeMapping>, std::vector<KmerIDX>> NodeMapper::map_kmers_to_graph(seqID_t id, std::vector<KmerIDX>& kmers) {
144  NodeMapping mapping;
145  mapping.initiate_mapping(id);
146  // For each kmer on sequence.
147  uint64_t mapped_kmers_count {0};
148  std::vector<NodeMapping> sequence_mappings;
149  std::vector<KmerIDX> unmapped_kmers;
150  for (auto &sk:kmers) {
151  bool found_kmer;
152  graphStrandPos graph_pos;
153  std::tie(found_kmer, graph_pos) = graph_kmer_index.find_unique_kmer_in_graph(sk.kmer);
154  // IF KMER EXISTS ON GRAPH
155  if (found_kmer) {
156  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Found kmer: " << sk.kmer << "in graph" << std::endl;
157  sglib::OutputLog(sglib::LogLevels::DEBUG) << '(' << sg.nodeID_to_name(graph_pos.node) << ", " << graph_pos.pos << ')' << std::endl;
158  mapped_kmers_count++;
159  // IF THE KMER MATCH IS THE FIRST MATCH FOR THE MAPPING...
160  if (!mapping.ismatched()) {
161  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Kmer match is the first for the mapping." << std::endl;
162  start_new_mapping(mapping, graph_pos, sk.pos);
163  } // IF THE KMER MATCH IS NOT THE FIRST MATCH FOR THE MAPPING...
164  else {
165  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Kmer match is not the first match for the mapping." << std::endl;
166  // THERE ARE TWO SITUATIONS WHERE WE WOULD START A NEW MAPPING...
167  // A). WE ARE MAPPING TO A COMPLETELY DIFFERENT NODE...
168  // B). THE DIRECTION OF THE MAPPING IS FLIPPED...
169 
170  if (!mapping.mapping_continues(graph_pos)) {
171  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Kmer match does not continues the current mapping." << std::endl;
172  sequence_mappings.push_back(mapping);
173  start_new_mapping(mapping, graph_pos, sk.pos);
174  } else { // IF NODE KMER MAPS TO IS THE SAME, EXTEND THE CURRENT MAPPING...
175  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Kmer match continues the current mapping." << std::endl;
176  mapping.extend(graph_pos.pos, sk.pos);
177  }
178  }
179  } else {
180  unmapped_kmers.emplace_back(sk.kmer);
181  }
182  }
183  // TODO : Check if this last push_back is required
184  if (mapping.ismatched()) sequence_mappings.push_back(mapping);
185  sglib::OutputLog(sglib::LogLevels::INFO) << "Mapped " << mapped_kmers_count << " kmers from the reference." << std::endl;
186  return std::make_tuple(std::move(sequence_mappings), std::move(unmapped_kmers));
187 }

300 bool NodeMapping::direction_will_continue(int32_t next_position) const {
301  auto direction = node_direction();
