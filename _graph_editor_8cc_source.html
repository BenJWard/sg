7 bool GraphEditor::detach_path(SequenceGraphPath p, bool consume_tips) {
8  SequenceGraph& sg(ws.getGraph());
9  if (p.getNodes().size()==1) return true;
10  //TODO: check the path is valid?
11  if (p.getNodes().size()==2) {
12  //std::cout<<"DETACHING path with only 2 nodes"<<std::endl;
13  if (p.getNodes().size()==2){
14  auto fwls=sg.get_fw_links(p.getNodes()[0]);
15  for (auto l:fwls) if (l.dest!=p.getNodes()[1]) sg.remove_link(l.source,l.dest);
16  auto bwls=sg.get_bw_links(p.getNodes()[1]);
17  for (auto l:bwls) if (l.dest!=-p.getNodes()[0]) sg.remove_link(l.source,l.dest);
18  }
19  return true;
20  }
21  //check if the path is already detached.
22  if (p.is_unitig()) return true;
23  //check if the nodes in the path have already been modified
24  for (auto n:p.getNodes()) {
25  if (edited_nodes.count(llabs(n))>0) {
26  std::cout<<"NOT DETACHING: path has already edited-out nodes"<<std::endl;
27  return false;
28  }
29  }
30  //TODO: create a new node with the sequence of the "middle" nodes
31  auto pmid=p;
32  pmid.getNodes().clear();
33  for (auto i=1;i<p.getNodes().size()-1;++i) pmid.getNodes().push_back(p.getNodes()[i]);
34  auto src=p.getNodes().front();
35  auto dest=p.getNodes().back();
36  if (!pmid.is_canonical()) {
37  pmid.reverse();
38  auto nsrc=-dest;
39  dest=-src;
40  src=nsrc;
41  }
42  sgNodeID_t new_node=sg.add_node(Node(pmid.get_sequence()));
43  //TODO: migrate connection from the src node to the first middle node into the new node
44  auto old_link=sg.get_link(-p.getNodes()[0],p.getNodes()[1]);
45  sg.remove_link(-p.getNodes()[0],p.getNodes()[1]);
46  sg.add_link(-src,new_node,old_link.dist);
47  //TODO: migrate connection from the last middle node to the dest node into the new node
48  old_link=sg.get_link(-p.getNodes()[p.getNodes().size()-2],p.getNodes()[p.getNodes().size()-1]);
49  sg.remove_link(-p.getNodes()[p.getNodes().size()-2],p.getNodes()[p.getNodes().size()-1]);
50  sg.add_link(-new_node,dest,old_link.dist);
51  //TODO: delete nodes from the original path while there is any of them with a disconnecte end.
52  bool mod=true;
53  std::set<sgNodeID_t> mid_nodes;
54  for (auto n:pmid.getNodes()) mid_nodes.insert(llabs(n));
55  while (mod){

57  for (auto n:mid_nodes) {
58  if (sg.get_fw_links(n).size()==0 or sg.get_bw_links(n).size()==0) {
59  mod=true;
60  sg.remove_node(n);
