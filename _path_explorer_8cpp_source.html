17 std::vector<SequenceGraphPath> PathExplorer::collect_paths(const sgNodeID_t seed,
18  const sgNodeID_t target,
19  const unsigned int size_limit,
20  const unsigned int edge_limit) const {
21 
22  struct visitor {
23  sgNodeID_t node;
24  uint dist;
25  uint path_length;
26  visitor(sgNodeID_t n, uint d, uint p) : node(n), dist(d), path_length(p) {}
27  };
28 
29  std::vector<SequenceGraphPath> collected_paths;
30  std::vector<sgNodeID_t> current_path;
31 
32  std::stack<visitor> to_visit;
33  to_visit.emplace(seed, 0, 0);
34  std::set<sgNodeID_t> visited;
35 
36  while (!to_visit.empty()) {
37  const auto activeNode(to_visit.top());
38  to_visit.pop();
39  if (visited.find(activeNode.node) == visited.end() and (activeNode.path_length < edge_limit or edge_limit == 0) and (activeNode.dist < size_limit or size_limit == 0) )
40  {
41  visited.emplace(activeNode.node);
42 
43  if (current_path.size() > activeNode.path_length) {
44  current_path[activeNode.path_length] = activeNode.node;
45  current_path.resize(activeNode.path_length + 1);
46  } else {
47  current_path.emplace_back(activeNode.node);
48  }
49 
50  for (const auto &l : sg.get_fw_links(activeNode.node)) {
51  // For each child of the current node, create a child visitor object, and enter it into the parent map.
52  visitor child { l.dest, activeNode.dist + uint(sg.nodes[std::abs(l.dest)].sequence.length() + l.dist), activeNode.path_length + 1 };
53 
54  // If I've found the target along this dfs path, I need to collect it and put it into a SequenceGraphPath object.
55  if (child.node == target) {
56  current_path.emplace_back(target);
57  collected_paths.emplace_back(sg, current_path);
58  } else {
59  to_visit.emplace(child);
60  }
61  }
62  }
63  }
64  return collected_paths;
65 }
