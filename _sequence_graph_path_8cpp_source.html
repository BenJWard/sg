25 std::string SequenceGraphPath::get_sequence() const {
26  sglib::OutputLog(sglib::LogLevels::DEBUG) << "get_sequence for a SequenceGraphPath with nodes = [" ;
27  if(sglib::OutputLogLevel == sglib::LogLevels::DEBUG) {
28  for(auto &n:nodes) std::cout << " " << n;
29  std::cout << " ]" << std::endl;
30  }
31  std::string s = "";
32  sgNodeID_t pnode = 0;
33  // just iterate over every node in path - contig names are converted to ids at construction
34  for (auto &n:nodes) {
35  std::string nseq;
36  if (n>0){
37  nseq = sg.nodes[n].sequence;
38  } else {
39  auto rcn = sg.nodes[-n];
40  rcn.make_rc();
41  nseq = rcn.sequence;
42  }
43  if (pnode !=0){
44  //find link between pnode' output (+pnode) and n's sink (-n)
45  auto l=sg.links[(pnode>0 ? pnode:-pnode)].begin();
46  for (;l!=sg.links[(pnode>0 ? pnode:-pnode)].end();++l)
47  if (l->source==pnode and l->dest==n) break;
48  if (l==sg.links[(pnode>0 ? pnode:-pnode)].end()) {
49  std::cout<<"can't find a link between "<<pnode<<" and "<<n<<std::endl;
50  throw std::runtime_error("path has no link");
51  } else {
52  if (l->dist > 0){
53  for (auto c = l->dist; c > 0; --c) s += "N";
54  }
55  else {
56  auto ovl =- l->dist;
57  for (auto s1 = s.c_str() + s.size() - ovl, s2 = nseq.c_str(); *s1 != NULL; ++s1, ++s2)
58  if (*s1 != *s2)
59  throw std::runtime_error("path overlap is invalid!");
60  nseq.erase(0, ovl);
61  }
62  }
63  }
64  s+=nseq;
65  pnode=-n;
66  }
67  sglib::OutputLog(sglib::LogLevels::DEBUG) << "get_sequence finished successfully for SequenceGraphPath with nodes = [" ;
68  if(sglib::OutputLogLevel == sglib::LogLevels::DEBUG) {
69  for(auto &n:nodes) std::cout<<" "<<n;
70  std::cout<<" ]"<<std::endl;
71  }
72  return s;
73 }

75 bool SequenceGraphPath::append_to_path(const sgNodeID_t newnode) {
76  if (nodes.empty()) {
77  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Path is new and empty." << std::endl;
78  nodes.emplace_back(newnode);
79  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Size of path is now: " << nodes.size() << std::endl;
80  return true;
81  }
82  sgNodeID_t from = -(nodes.back());
83  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Path is not new and empty: trying to append." << std::endl;
84  bool append_possible = sg.link_exists(from, newnode);
85  if (append_possible) {
86  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Append is possible, adding node to the back." << std::endl;
87  nodes.emplace_back(newnode);
88  sglib::OutputLog(sglib::LogLevels::DEBUG) << "Size of path is now: " << nodes.size() << std::endl;
89  }
90  return append_possible;
91 }
