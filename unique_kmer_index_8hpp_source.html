38  void generate_index(const SequenceGraph &sg, uint8_t k) {
39  std::vector<KmerIDX> kidxv;
40  uint64_t total_k { 0 };
41  total_kmers_per_node = std::vector<uint64_t>(sg.nodes.size(), 0);
42  for (sgNodeID_t node = 0; node < sg.nodes.size(); node++) {
43  auto sgnode = sg.nodes[node];
44  if (sgnode.sequence.size() >= k) {
45  auto n = sgnode.sequence.size() + 1 - k;
46  total_k += n;
47  total_kmers_per_node[node] = n;
48  }
49  }
50  kidxv.reserve(total_k);
51  FastaRecord r;
52  kmerIDXFactory<FastaRecord> kcf({k});
53  for (sgNodeID_t n = 1; n < sg.nodes.size(); ++n) {
54  if (sg.nodes[n].sequence.size() >= k) {
55  r.id = n;
56  r.seq = sg.nodes[n].sequence;
57  kcf.setFileRecord(r);
58  kcf.next_element(kidxv);
59  }
60  }
61  sglib::OutputLog(sglib::INFO) << kidxv.size() << " kmers in total" << std::endl;
62  sglib::OutputLog(sglib::INFO) << " Sorting..." << std::endl;
63  std::sort(kidxv.begin(), kidxv.end());
64 
65  sglib::OutputLog(sglib::INFO) << " Merging..." << std::endl;
66  auto wi = kidxv.begin();
67  auto ri = kidxv.begin();
68  auto nri = kidxv.begin();
69  while (ri < kidxv.end()) {
70  //std::cout << "ri kmer: " << ri -> kmer << std::endl;
71  while (nri != kidxv.end() and nri -> kmer == ri -> kmer) {
72  //std::cout << "nri kmer: " << nri -> kmer << std::endl;
73  ++nri;
74  }
75  //std::cout << "kmer appears " << nri - ri << " time(s)" << std::endl;
76  if (nri - ri == 1) {
77  *wi = *ri;
78  ++wi;
79  }
80  ri = nri;
81  }
82  kidxv.resize(wi - kidxv.begin());
83  sglib::OutputLog(sglib::INFO) << kidxv.size() << " unique kmers in index, creating map" << std::endl;
84  std::unordered_set<int32_t> seen_contigs;
85  seen_contigs.reserve(sg.nodes.size());
86  unique_kmers_per_node = std::vector<uint64_t>(sg.nodes.size(), 0);
87  for (auto &kidx :kidxv) {
88  kmer_to_graphposition[kidx.kmer] = { kidx.contigID, kidx.pos };
89  unique_kmers_per_node[std::abs(kidx.contigID)] += 1;
90  seen_contigs.insert((kidx.contigID > 0 ? kidx.contigID : -kidx.contigID));
91  }
92  sglib::OutputLog(sglib::INFO) << seen_contigs.size() << " nodes with indexed kmers" <<std::endl;
93  }

191  void read_from_file(std::string filename) {
192  sglib::OutputLog() << "Loading index" << std::endl;
193  std::ifstream inf(filename, std::ios_base::binary);
194  pair skm{0, graphStrandPos(0, 0)};
195  inf.read(reinterpret_cast<char *>(&k), sizeof(k));
196  auto mapSize(kmer_to_graphposition.size());
197  inf.read(reinterpret_cast<char *>(&mapSize), sizeof(mapSize));
198  kmer_to_graphposition.reserve(mapSize);
199  for (decltype(kmer_to_graphposition.size()) n = 0; n < mapSize; ++n){
200  inf.read(reinterpret_cast<char *>(&skm.first), sizeof(skm.first));
201  inf.read(reinterpret_cast<char *>(&skm.second), sizeof(skm.second));
202  kmer_to_graphposition.emplace(skm.first, skm.second);
203  }
204 
205  read_vector(inf, unique_kmers_per_node);
206  read_vector(inf, total_kmers_per_node);
207 
208  sglib::OutputLog() << "Done!" << std::endl;
