163 std::vector<sgNodeID_t> WorkSpace::select_from_all_nodes(uint32_t min_size, uint32_t max_size, uint32_t min_tags, uint32_t max_tags,
164  float min_ci, float max_ci) {
165  std::vector<sgNodeID_t> nodes;
166  sglib::OutputLog()<<"Selecting nodes: " << min_size << "-" << max_size << " bp " <<
167  min_tags << "-" << max_tags << " tags " << min_ci << "-" << max_ci << " CI"<<std::endl;
168  uint64_t tnodes=0,tbp=0;
169  nodes.reserve(sg.nodes.size());
170 #pragma omp parallel
171  {
172  std::vector<sgNodeID_t> thread_nodes;
173  uint64_t ttbp=0;
174 #pragma omp for schedule(static, 100)
175  for (auto n=1;n<sg.nodes.size();++n) {
176  if (sg.nodes[n].sequence.size() < min_size) continue;
177  if (sg.nodes[n].sequence.size() > max_size) continue;
178  if (!linked_read_mappers.empty()) {
179  auto ntags = linked_read_mappers[0].get_node_tags(n);
180  if (ntags.size() < min_tags or ntags.size() > max_tags) continue;
181  }
182  if (!kci.graph_kmers.empty()) {
183  auto ci = kci.compute_compression_for_node(n, 1);
184  if (ci < min_ci or ci > max_ci) continue;
185  }
186  ++tnodes;
187  thread_nodes.emplace_back(n);
188 
189  ttbp += sg.nodes[n].sequence.size();
190  }
191 
192 #pragma omp critical(collect_selected_nodes)
193  {
194  nodes.insert(nodes.end(), thread_nodes.begin(), thread_nodes.end());
195  tbp+=ttbp;
196  }
197 
198  }
199  sglib::OutputLog()<< "Selected "<<tnodes<<" / "<<sg.nodes.size()<<" with a total "<<tbp<<"bp"<< std::endl;
200  return nodes;
201 }
