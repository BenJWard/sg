214 std::vector<uint64_t> PairedReadMapper::size_distribution() {
215  frdist.clear();
216  frdist.resize(20000);
217  rfdist.clear();
218  rfdist.resize(20000);
219  uint64_t frcount=0,rfcount=0;
220  std::vector<int32_t> read_firstpos(read_to_node.size()),read_lastpos(read_to_node.size());
221  std::vector<bool> read_rev(read_to_node.size());
222  for (auto n=1;n<sg.nodes.size();++n) {
223  for (auto &rm:reads_in_node[n]) {
224  read_firstpos[rm.read_id]=rm.first_pos;
225  read_lastpos[rm.read_id]=rm.last_pos;
226  read_rev[rm.read_id]=rm.rev;
227  }
228  }
229  for (uint64_t r1=1;r1<read_to_node.size();r1+=2){
230  if (read_to_node[r1]!=0 and read_to_node[r1]==read_to_node[r1+1]) {
231  auto node=read_to_node[r1];
232  ReadMapping rm1,rm2;
233  rm1.first_pos=read_firstpos[r1];
234  rm1.last_pos=read_lastpos[r1];
235  rm1.rev=read_rev[r1];
236  rm2.first_pos=read_firstpos[r1+1];
237  rm2.last_pos=read_lastpos[r1+1];
238  rm2.rev=read_rev[r1+1];
239  if (rm1.first_pos>rm2.first_pos) std::swap(rm1,rm2);
240  auto d=rm2.last_pos-rm1.first_pos;
241  if (d>=rfdist.size()) continue;
242  if (rm1.rev and !rm2.rev) {
243  ++rfdist[d];
244  ++rfcount;
245  }
246  if (!rm1.rev and rm2.rev) {
247  ++frdist[d];
248  ++frcount;
249  }
250  }
251  }
252  //std::cout<<"Read orientations: FR: "<<frcount<<" RF: "<<rfcount<<std::endl;
253  if (frcount>rfcount){
254  return frdist;
255  } else return rfdist;
256 }
