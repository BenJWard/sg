22  uint64_t hash(uint64_t key) {
23  key = (~key + (key << 21)) & mask; // key = (key << 21) - key - 1;
24  key = key ^ key >> 24;
25  key = ((key + (key << 3)) + (key << 8)) & mask; // key * 265
26  key = key ^ key >> 14;
27  key = ((key + (key << 2)) + (key << 4)) & mask; // key * 21
28  key = key ^ key >> 28;
29  key = (key + (key << 31)) & mask;
30  return key;
31  }

44  inline std::vector<MinPosIDX>::size_type getMinSketch(const std::string &seq, std::vector<MinPosIDX> &sketch){
45  // TODO: Adjust for when K is larger than what fits in uint64_t!
46  last_unknown=0;
47  fkmer=0;
48  rkmer=0;
49  sketch.reserve(seq.length()/w);
50  if (seq.length() < K) return sketch.size();
51 
52  for (unsigned int nt = 0; nt < K; nt++) {
53  fillKBuf(seq[nt], fkmer, rkmer, last_unknown);
54  }
55 
56  int32_t pos = K;
57  for (; pos < seq.length()-w+1; ++pos) {
58  uint64_t min(std::numeric_limits<uint64_t>::max());
59  uint64_t pFkmer(fkmer);
60  uint64_t pRkmer(rkmer);
61  for (unsigned int j = 0; j < w; j++) {
62  fillKBuf(seq[pos + j], fkmer, rkmer, last_unknown);
63  if (fkmer == rkmer) continue;
64  if (fkmer < rkmer) {
65  // Is fwd
66  min = fkmer < min ? fkmer : min;
67  } else {
68  // Is bwd
69  min = rkmer < min ? rkmer : min;
70  }
71  }
72  fkmer = pFkmer;
73  rkmer = pRkmer;
74  for (unsigned int j = 0; j < w; j++) {
75  fillKBuf(seq[pos + j], fkmer, rkmer, last_unknown);
76  if (fkmer < rkmer and min == fkmer) {
77  // Is fwd
78  sketch.emplace_back(hash(min), pos+j);
79  continue;
80  }
81  if (rkmer < fkmer and min == rkmer) {
82  // Is bwd
83  sketch.emplace_back(hash(min), -1*(pos+j));
84  }
85  }
86  }
87  return sketch.size();
88  }
